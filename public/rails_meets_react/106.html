is called after each successful re-render of a component. It is

for the first time a component is rendered. That means it is usually called after re-rendering

In both of these methods, you have an access to

componentDidMount is great for integrating data sources.
CocktailList = React.createClass
displayName: 'CocktailList'
getInitialState: ->
cocktails: []
cocktailCategories: []
cocktailsGroupedByCategory: ->
groupedItems = {}
for cocktailCategory in @state.cocktailCategories
groupedItems[cocktailCategory.id] = []
for cocktail in @state.cocktails
groupedItems[cocktail.categoryID].push(cocktail)
groupedItems
componentDidMount: ->
{eventBus} = @props
eventBus.onValue (event) =>
switch event.type
when 'cocktailsLoaded', 'cocktailsUpdated'
@setCocktails(event.cocktails)
when 'cocktailCategoriesLoaded', 'cocktailCategoriesUpdated'
@setCocktailCategories(event.cocktailCategories)
setCocktails: (cocktails) ->
@setState cocktails: cocktails
setCocktailCategories: (cocktailCategories) ->
@setState cocktailCategories: cocktailCategories

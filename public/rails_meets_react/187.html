d)} toolbox:"
React.DOM.button(
key: 'show-hide-button',
@buttonLabels()[@state.hidden],
onClick: => @setState hidden: !@state.hidden
)
)
toolsList(key: 'list', tools: @props.toolbox.tools) unless @state.hidden
verificationStatus: (verified) ->
if verified
"verified"
else
"unverified"

Filtering chat channels - performed at most 4 times per second so the list is not refreshed after every quick
keystroke
ChannelSearch = React.createClass
getInitialState: ->
searchQuery: ''
componentWillMount: ->
@changeSearchQuery = _.debounce(@changeSearchQuery, 250)
changeSearchQuery: (event) ->
newSearchQuery = event.target.value
@setState(
searchQuery: newSearchQuery
)
filteredChannels: ->
@props.channels.filter (channel) =>
channel.indexOf(@state.searchQuery) == 0
render: ->
React.DOM.div null,
React.DOM.input
type: 'text'
onChange: @changeSearchQuery
channelsList
channels: @filteredChannels()
ChannelsList = React.createClass
render: ->
React.DOM.div null,
@props.channels.map((channel) ->
React.DOM.div
key: channel
React.DOM.input
type: 'text'
value: channel
readOnly: true
React.DOM.br
)
